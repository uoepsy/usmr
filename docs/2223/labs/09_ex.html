df %>% mutate(
  season = case_when(
    birthmonth %in% tolower(month.abb)[c(1,2,12)] ~ "winter",
    birthmonth %in% tolower(month.abb)[3:5] ~ "spring",
    birthmonth %in% tolower(month.abb)[6:8] ~ "summer",
    birthmonth %in% tolower(month.abb)[9:11] ~ "autumn",
    TRUE ~ NA_character_
  ),
  season = factor(season)
) -> df
